Sorting Algorithm Examples

Example 1: Bubble Sort - The Brute Force Approach


# Simple implementation of Bubble Sort
def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n-i-1):
            if arr[j] > arr[j+1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
    return arr

# Example usage:
array = [64, 34, 25, 12, 22]
sorted_array = bubble_sort(array.copy())  # [12, 22, 25, 34, 64]

Initial thinking: Compare adjacent elements and swap if they're in the wrong order. Simple but inefficient for large datasets.

Example 2: Quick Sort - The Efficient Approach


# Efficient implementation of Quick Sort
def quick_sort(arr):
    if len(arr) <= 1:
        return arr
    
    pivot = arr[len(arr) // 2]
    left = [x for x in arr if x < pivot]
    middle = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    
    return quick_sort(left) + middle + quick_sort(right)

# Example usage:
array = [64, 34, 25, 12, 22]
sorted_array = quick_sort(array.copy())  # [12, 22, 25, 34, 64]

Improved thinking: Use divide-and-conquer to partition the array around a pivot, reducing the problem size efficiently.

Example 3: Merge Sort - The Stable Approach


# Stable implementation of Merge Sort
def merge_sort(arr):
    if len(arr) <= 1:
        return arr
    
    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])
    return merge(left, right)

def merge(left, right):
    result = []
    i = j = 0
    
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:  # <= makes it stable
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1
    
    result.extend(left[i:])
    result.extend(right[j:])
    return result

# Example usage:
array = [64, 34, 25, 12, 22]
sorted_array = merge_sort(array.copy())  # [12, 22, 25, 34, 64]

Systematic thinking: Break the problem into smaller sorted arrays and merge them while maintaining order.
